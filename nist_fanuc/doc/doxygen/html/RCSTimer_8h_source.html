138  int wait ( )
139  {
140  // boost::chrono::high_resolution_clock::time_point interval; /* interval between this and last wakeup */
141  double numcycles; /* interval, in units of timeout */
142  int missed = 0; /* cycles missed */
143  double remaining = 0.0; /* time remaining until timeout */
144  boost::chrono::high_resolution_clock::time_point time_in; /* time wait() was entered */
145  boost::chrono::high_resolution_clock::time_point time_done; /* time user function finished */
146 
147  if ( _bSuspend )
148  {
149  boost::mutex::scoped_lock aLock(condMutex);
150  cWakeup.wait(aLock);
151  }
152 
153  /* first call the user timing function, if any */
154  if ( function != 0 )
155  {
156  /* set time in */
157  time_in = etime( );
158 
159  if ( ( *function )( 0 ) == -1 )
160  {
161  return -1; /* fatal error in timing function */
162  }
163  time_done = etime( );
164  }
165  else
166  {
167  /* set time in, time done not used */
168  time_in = etime( );
169  }
170 
171  /* calculate the interval-- for user timing functions, this is how
172  long between this wakeup and the last wakeup. For internal timers,
173  this is how long we need to sleep to make it to the next interval
174  on time. */
175  double interval = ToNanoseconds(time_in - last_time) / 1E9;
176  numcycles = interval / timeout;
177 
178  /* synchronize and set last_time correctly; update idle time */
179  counts++;
180 
181  if ( function != 0 )
182  {
183  last_time = time_done;
184  }
185  idle += timeout > interval ? timeout - interval : 0;
186  busy += interval;
187  missed = (int) numcycles;
188  remaining = timeout - interval;
189  esleep(remaining);
190  last_time = etime( );
191  return missed;
192  }
