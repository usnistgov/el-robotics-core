421 inline EigenPosition EPoseMult (const EigenPose & p, const EigenPosition & v)
422 {
423  double ww = 2 * p(4) * p(3);
424  double wx = 2 * p(3) * p(4);
425  double wy = 2 * p(3) * p(5);
426  double wz = 2 * p(4) * p(6);
427  double xx = 2 * p(4) * p(4);
428  double xy = 2 * p(4) * p(5);
429  double xz = 2 * p(4) * p(6);
430  double yy = 2 * p(5) * p(5);
431  double yz = 2 * p(5) * p(6);
432  EigenPosition vnew;
433 
434  vnew(0) = ( 1 - xx - yy ) * v(0) + ( wx - yz ) * v(1) + ( wy + xz ) * v(2) + p(0);
435  vnew(1) = ( wx + yz ) * v(0) + ( 1 - ww - yy ) * v(1) + ( xy - wz ) * v(2) + p(1);
436  vnew(2) = ( wy - xz ) * v(0) + ( xy + wz ) * v(1) + ( 1 - ww - xx ) * v(2) + p(2);
437  return vnew;
438 }

489 inline EigenPosition EQuatCartMult (Eigen::Quaterniond & q1, const EigenPosition v2)
490 {
491  EigenPosition c;
492  EigenPosition vout(3);
493 
494  if ( !EQuatIsNorm(q1) )
495  {

497  }
498  c.resize(3);

501  c(2) = q1.x() * v2(1) - q1.y() * v2(0);
502

505  vout(1) = v2(1) + 2 * ( q1.w() * c(1) + q1.z() * c(0) - q1.x() * c(2) );
506  vout(2) = v2(2) + 2 * ( q1.w() * c(2) + q1.x() * c(1) - q1.y() * c(0) );
507 
508  return vout;

548 inline Eigen::Quaterniond EQuatFromRpy (double & roll, double & pitch, double & yaw) // Assuming the angles are in radians.
549 {
550  Eigen::Quaterniond q;
551  double c1 = cos(roll / 2);
552  double s1 = sin(roll / 2);
553  double c2 = cos(pitch / 2);
554  double s2 = sin(pitch / 2);
555  double c3 = cos(yaw / 2);
556  double s3 = sin(yaw / 2);
557  double c1c2 = c1 * c2;
558  double s1s2 = s1 * s2;
559 
560  q.w() = c1c2 * c3 - s1s2 * s3;
561  q.x() = c1c2 * s3 + s1s2 * c3;
562  q.y() = s1 * c2 * c3 + c1 * s2 * s3;
563  q.z() = c1 * s2 * c3 - s1 * c2 * s3;
564  return q;
565 }
571 inline void EQuatToRpy (const Eigen::Quaterniond & q1, double & roll, double & pitch, double & yaw)
572 {
573  double sqw;
574  double sqx;
575  double sqy;
576  double sqz;
577 
578  sqx = q1.x() * q1.x();
579  sqy = q1.y() * q1.y();
580  sqz = q1.z() * q1.z();
581  sqw = q1.w() * q1.w();
582 
583  // Cases derived from https://orbitalstation.wordpress.com/tag/quaternion/
584  double sarg = -2 * ( q1.x() * q1.z() - q1.w() * q1.y() );
585 
586  if ( sarg <= -0.99999 )
587  {
588  pitch = -0.5 * M_PI;
589  roll = 0;
590  yaw = 2 * atan2(q1.x(), -q1.y());
591  }
592  else if ( sarg >= 0.99999 )
593  {
594  pitch = 0.5 * M_PI;
595  roll = 0;
596  yaw = 2 * atan2(-q1.x(), q1.y());
597  }
598  else
599  {
600  pitch = asin(sarg);
601  roll = atan2(2 * ( q1.y() * q1.z() + q1.w() * q1.x() ), sqw - sqx - sqy + sqz);
602  yaw = atan2(2 * ( q1.x() * q1.y() + q1.w() * q1.z() ), sqw + sqx - sqy - sqz);
603  }
604 }
